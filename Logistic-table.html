2.1 简略版

R code 
library(tidyverse)
library(gtsummary)
data <- trial
# 单因素
uvreg1 <-
  data |>
  select(trt,age,grade,response) |>
  tbl_uvregression(
    y = response, #因变量
    include = c(trt,age,grade), # 自变量
    method = glm, #选用对应的模型方法
    method.args = list(family = binomial), #选用对应的模型参数
    exponentiate = T,
    hide_n = T
  ) |>
  bold_labels() |>
  italicize_levels() |>
  bold_p(t = 0.10) 

# 多因素
model1 <-
  reformulate(
    data |> select(trt,age,grade) |> colnames(),
    response = "I(response == '1')"
  )

mureg1 <-
  glm(model1,family = binomial,data = data) |>
  tbl_regression(
    exponentiate = T
  ) |>
  bold_p(t = 0.05) 

# reg1 <- # 2个table进行横向合并
  tbl_merge(
    tbls = list(uvreg1, mureg1),
    tab_spanner = c("**Univariable**", "**Multivariable**")
  )
